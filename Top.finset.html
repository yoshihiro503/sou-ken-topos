Variable U : Type.
Inductive Finite : Ensemble U → Prop :=
| Empty_is_finite : Finite (Empty_set U)
| Union_is_finite : ∀ A : Ensemble U, Finite A → ∀ x : U, ¬ In U A x → Finite (Add U A x).
Inductive cardinal : Ensemble U → nat → Prop :=
| card_empty : cardinal (Empty_set U) 0
| card_add : ∀ (A : Ensemble U) (n : nat), cardinal A n → ∀ x : U, ¬ In U A x → cardinal (Add U A x) (S n).
